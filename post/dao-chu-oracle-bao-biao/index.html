导出 Oracle 报表

其他
2017-06-30
  1. 登入oracle

sqlplus niosdb/Niosoptr@jxwydb

  1. 导入sql

@/home/oracle/zqk/zhibiaocity.sql
ql内添加分隔>||','||

sqlplus  niosdb/Niosoptr@jxwydb <<EOF
@/home/oracle/zqk/zhibiaocity.sql
EOF
  1. 添加表头
sed '1i\
name1,name2,name2...' 1.csv>2.csv
  1. 整合脚本
#!/bin/ksh

cd /home/oracle/zqk/cellflow
putfile="aaa.csv"

sqlplus  niosdb/Niosoptr@jxwydb <<EOF
@/home/oracle/zqk/cellflow/cellflow.sql
EOF

sed '1i\
时间（小时级),ECGI,CELLNAME,上行流量,下行流量' $putfile>tmp.file

putfile="cellflow0515.csv"

cp tmp.file $putfile
rm tmp.file
rm aaa.csv
